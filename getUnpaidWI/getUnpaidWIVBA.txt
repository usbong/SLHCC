Sub UnpaidWI()
Dim UnpaidWI2017 As Worksheet
Dim output As Worksheet
Dim row As Range
Dim row2 As Range
Dim RowCount As Integer
Dim RowCountOutput As Integer

RowCount = 1
RowCountOutput = 1

Set UnpaidWI2017 = Sheets("unpaidWI2017")
Set output = Sheets("unpaidWI2017Output")

For Each row In UnpaidWI2017.Rows
    If ((UnpaidWI2017.Cells(RowCount, 2).Value <> "Mike") And ((UnpaidWI2017.Cells(RowCount, 14).Value = "SC") Or (UnpaidWI2017.Cells(RowCount, 14).Value = "WI") Or (UnpaidWI2017.Cells(RowCount, 14).Value = "SLC") Or (UnpaidWI2017.Cells(RowCount, 14).Value = "PWD"))) Then
        
       UnpaidWI2017.Range("A" & RowCount, "CB" & RowCount).Columns.Copy
       
       output.Select
       Rows(RowCountOutput).PasteSpecial
       
       'remove blank space
       RowCountOutput = RowCountOutput + 1
    End If

    If RowCount = 1370 Then
      Exit For
    Else
      RowCount = RowCount + 1
    End If
Next row

Range("D1").Value = 1
Range("F1").Value = 1

MsgBox (RowCount)

End Sub
